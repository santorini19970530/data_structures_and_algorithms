funciton Thing(n)
  if n < 3 then
    return 1
  end if

  new Stack s

  PUSH[1, s]
  PUSH[1, s]

  for 3 <= i <= n do
    x = TOP[s]
    POP[s]
    y = TOP[s]
    PUSH[x, s]
    PUSH[x + y, s]
  end for

  return TOP[s]
end function
